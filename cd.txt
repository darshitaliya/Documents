1) Character Count Framing Technique

def character_count(data):
    frames = []
    words = data.split()
    for word in words:
        count = len(word) + 1
        frame = str(count) + word
        frames.append(frame)
    return frames

data = input("Enter data: ")
result = character_count(data)
print("Frames:", result)


2) TCP Client Server Communication

*server
import socket
s = socket.socket()
s.bind(('localhost', 12345))
s.listen(1)
print("Server waiting...")
conn, addr = s.accept()
msg = conn.recv(1024).decode()
print("Client:", msg)
conn.send("Hello Client".encode())
conn.close()
s.close()

*client
import socket
c = socket.socket()
c.connect(('localhost', 12345))
c.send("Hello Server".encode())
msg = c.recv(1024).decode()
print("Server:", msg)
c.close()

3) UDP Client Server Communication

*server
import socket
server = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
server.bind(('localhost', 12345))
print("UDP Server waiting...")
data, addr = server.recvfrom(1024)
print("Client:", data.decode())
server.sendto("Hello Client".encode(), addr)

*client
import socket
client = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
client.sendto("Hello Server".encode(), ('localhost', 12345))
data, addr = client.recvfrom(1024)
print("Server:", data.decode())


4) File Transfer using TCP

*server
import socket

server = socket.socket()
server.bind(('localhost', 12345))
server.listen(1)

print("Server waiting...")

conn, addr = server.accept()

file = open("sample2.txt", "rb")

data = file.read(1024)
while data:
    conn.send(data)
    data = file.read(1024)

file.close()
conn.close()
server.close()

print("File sent successfully")

*client
import socket

client = socket.socket()
client.connect(('localhost', 12345))

file = open("received.txt", "wb")

data = client.recv(1024)
while data:
    file.write(data)
    data = client.recv(1024)

file.close()
client.close()

print("File received successfully")

5) Bit Stuffing

data=input("Enter bit stream: ")
count=0
result=""
for bit in data:
    if bit=='1':
        count+=1
    else:
        count=0
    result+=bit
    if count==5:
        result+='0'
        count=0
print(result)


6)Byte Stuffing

flag='F'
escape='E'
data=input("Enter data: ")
stuffed=""
for char in data:
    if char==flag or char==escape:
        stuffed+=escape
    stuffed+=char
stuffed=flag+stuffed+flag
print(stuffed)
